# %%
from io import StringIO
from itertools import takewhile
from typing import Generator

import pandas as pd
import requests
from toolz import curry, pipe

IDBR_URL = "https://www.nomisweb.co.uk/api/v01/dataset/NM_141_1.data.csv"

# %%
# 
def idbr_payload():

    fields = [
        "date",
        "geography_name",
        "geography_code",
        "industry_name",
        "industry_code",
        "industry_type",
        "employment_sizeband_name",
        "legal_status_name",
        "measures_name",
        "obs_value",
        "obs_status_name",
        "record_count",
    ]

    return {
        # local authorities: district / unitary (as of April 2021) within United Kingdom
        "geography": "TYPE432",
        # total of all employee sizebands
        "employment_sizeband": 0,
        # count of businesses
        "measures": 20100,
        # total of all legal statues
        "legal_status": 0,
        # year
        "time": 2021,
        # annual frequency
        "freq": "A",
        # SIC 5 industry level
        "industry": "TYPE32",
        # Column selection
        "select": ",".join(fields),
        # How many rows to return with each query (25K limit)
        "RecordLimit": 25000,
    }


# %%
# 
def idbr_pagination(endpoint: str, params: dict) -> Generator:
    """Yield paginated responses from IDBR API without termination."""
    record_offset = 0
    with requests.Session() as s:
        while True:
            r = s.get(
                endpoint,
                params={
                    **params,
                    "RecordOffset": record_offset,
                    # "ExcludeColumnHeadings": False if record_offset == 0 else True,
                },
                stream=True,
            )
            r.raise_for_status()
            yield r
            record_offset += params["RecordLimit"]


# %%


takewhile_ = curry(takewhile)
response_to_df = lambda response: pd.read_csv(
    StringIO(response.text), dtypes={"INDUSTRY_CODE": str}
)

out = pipe(
    idbr_pagination(IDBR_URL, idbr_payload()),
    map(response_to_df),
    takewhile_(lambda df: df.shape[0] != 0),
    pd.concat,
)
