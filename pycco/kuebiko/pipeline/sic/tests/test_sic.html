    print(out)
    print(clean_df)
    assert out.isna().sum().sum() == 0

def test_normalise_codes(clean_df):
    assert (
        normalise_codes(clean_df.loc[:, LEVELS])
        # Check for dots/slashes in `LEVELS`
        .apply(lambda col: (~col.str.contains(r"\.|/", regex=True)).all()).all()
    )
