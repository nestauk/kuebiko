import json

import toolz.curried as t
import tqdm

from perspex import PROJECT_DIR
from perspex.getters.inputs.url import get_urls
from perspex.pipeline.scraper.selenium import (
    DriverContainer,
    get_with_retry,
)
from perspex.pipeline.scraper.utils import (
    dismiss_alerts,
    extract,
    get_done_urls,
    page_stats,
    wait_for_readystate_complete,
)

# %%

with open(PROJECT_DIR / "outputs/data/scrape_.jsonl", "a+") as f:
    done_urls = get_done_urls(f)

    urls = t.pipe(
        get_urls(),
        t.filter(lambda url: str(url) not in done_urls),
        t.take(1_000),
        list,
    )

    print(f"Remaining {len(urls)} | Done {len(done_urls)}")

    with DriverContainer() as driver_container:
        out = t.pipe(
            urls,
            t.map(
                lambda url: (
                    str(url),
                    *extract(
                        get_with_retry(
                            driver_container,
                            url,
                            callback=t.compose_left(
                                dismiss_alerts, wait_for_readystate_complete
                            ),
                        )
                    ),
                )
            ),
            t.map(t.do(lambda row: f.write(json.dumps(row) + "\n"))),
            tqdm.tqdm,
            list,
        )

# Length of all (fail is 0 length)
t.pipe(
    out,
    t.map(lambda x: x[1] or ""),
    t.map(len),
    list,
    print,
)
