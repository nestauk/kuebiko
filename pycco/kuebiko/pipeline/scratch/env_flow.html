import os

from metaflow import current, environment, FlowSpec, Parameter, project, step
# 
@project(name="scratch")
class ScratchFlow(FlowSpec):
    @step
    def start(self):
        print(os.environ.get("s3_secret"))

        self.next(self.end)

    @environment(vars={"s3_secret": "set by decorator"})
    @step
    def end(self):
        print(os.environ.get("s3_secret"))


if __name__ == "__main__":
    ScratchFlow()
